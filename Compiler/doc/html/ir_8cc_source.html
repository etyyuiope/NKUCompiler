105 void compiler::ir::IR::emit_ir(std::ostream& output, const bool& verbose) {
106  // Wrap std::ostream in a macro.
107  if (type == ir::op_type::BEGIN_FUNC || type == ir::op_type::END_FUNC) {
108  FORMAT(output, op_name[type], 0x10);
109  } else if (type != ir::op_type::LBL) {
110  FORMAT(output, std::string(" ") + op_name[type], 0x10);
111  }
112 
113  // Emit IR of each operand. Deepmost callee.
114  auto lambda_walk_ir = [&output, this](Operand* const operand) {
115  if (operand != nullptr) {
116  if (operand->get_type() == var_type::NONE) {
117  FORMAT(output, operand->get_identifier(), 10);
118  } else if (operand->get_is_var() == false) {
119  FORMAT(output,
120  var_type_to_string(operand->get_type()) + " " +
121  operand->get_value(),
122  10);
123  } else if (operand->get_is_var() == true) {
124  FORMAT(output,
125  var_type_to_string(operand->get_type()) + " " +
126  operand->get_identifier(),
127  10);
128  }
129  }
130  };
131  walk_ir(lambda_walk_ir);
132 
133  FORMAT(output, label + "\n", 0);
134 }

207 compiler::ir::Operand* compiler::ir::dump_value(
208  compiler::Item_literal* const value) {
209  switch (value->get_literal_type()) {
210  case Item_literal::INT_TYPE: {
211  compiler::Item_literal_int* const val_int =
212  static_cast<compiler::Item_literal_int* const>(value);
213  return new compiler::ir::Operand(ir::var_type::i32, "",
214  std::to_string(val_int->get_int()),
215  false, false);
216  }
217 
218  case Item_literal::REAL_TYPE: {
219  compiler::Item_literal_real* const val_real =
220  static_cast<compiler::Item_literal_real* const>(value);
221  return new compiler::ir::Operand(
222  ir::var_type::DB, "", ir::convert_from_double(val_real->get_double()),
223  false, false);
224  }
225 
226  case Item_literal::CHAR_TYPE: {
227  compiler::Item_literal_char* const val_char =
228  static_cast<compiler::Item_literal_char* const>(value);
229  return new compiler::ir::Operand(
230  ir::var_type::i8, "", std::to_string((int)val_char->get_char()),
231  false, false);
232  }
233 
234  default:
235  throw compiler::unsupported_operation(
236  "Cannot dump operand from type" +
237  std::to_string(value->get_literal_type()));
238  }
239 }
