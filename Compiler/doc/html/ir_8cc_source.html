105void compiler::ir::IR::emit_ir(std::ostream& output, const bool& verbose) {
106 // Wrap std::ostream in a macro.
107 FORMAT(output, op_name[type]);
108
109 // Emit IR of each operand. Deepmost callee.
110 auto lambda_walk_ir = [&output, this](Operand* const operand) {
111 if (operand != nullptr) {
112 if (operand->get_type() == var_type::NONE) {
113 output << "\t";
114 } else if (operand->get_is_var() == false) {
115 output << var_type_to_string(operand->get_type()) << ' '
116 << operand->get_value() << "\t";
117 } else if (operand->get_is_var() == true) {
118 output << var_type_to_string(operand->get_type()) << ' '
119 << operand->get_identifier() << "\t";
120 }
121 }
122 };
123 walk_ir(lambda_walk_ir);
124
125 output << "\t" << label << std::endl;
126}

145bool compiler::ir::IR::emit_helper(

147 const bool& chained) const {
148 // This callback function is what we created in function walk_ir and the
149 // latter one is created in iterate_operand.
150 return (chained && callback(dst)) || callback(operand_a) ||
151 callback(operand_b) || callback(operand_c);

158bool compiler::ir::check_valid_binary(compiler::ir::Operand* const lhs,
159 compiler::ir::Operand* const rhs) {
160 if (lhs->get_type() == compiler::ir::var_type::NONE ||
161 lhs->get_type() == compiler::ir::var_type::NONE) {
162 return false;
163 } else if ((lhs->get_is_ptr() & rhs->get_is_var()) == 0) {
164 return false;
165 }
166 return true;

181std::string compiler::ir::var_type_to_string(
182 const compiler::ir::var_type& type) {
183 switch (type) {
184 case var_type::DB:
185 return "DOUBLE";
186 case var_type::i32:
187 return "i32";
188 case var_type::i8:
189 return "i8";
190 case var_type::NONE:
191 return "";
192 default:
193 throw compiler::unimplemented_error(
194 "This type of variable is not yet supported!");
195 }
196}

198compiler::ir::Operand* compiler::ir::dump_value(
199 compiler::Item_literal* const value) {
200 switch (value->get_literal_type()) {
201 case Item_literal::INT_TYPE: {
202 compiler::Item_literal_int* const val_int =
203 static_cast<compiler::Item_literal_int* const>(value);
204 return new compiler::ir::Operand(ir::var_type::i32, "",
205 std::to_string(val_int->get_int()),
206 false, false);
207 }
208
209 case Item_literal::REAL_TYPE: {
210 compiler::Item_literal_real* const val_real =
211 static_cast<compiler::Item_literal_real* const>(value);
212 return new compiler::ir::Operand(ir::var_type::DB, "",
213 std::to_string(val_real->get_double()),
214 false, false);
215 }
216
217 case Item_literal::CHAR_TYPE: {
218 compiler::Item_literal_char* const val_char =
219 static_cast<compiler::Item_literal_char* const>(value);
220 return new compiler::ir::Operand(
221 ir::var_type::i8, "", std::to_string((int)val_char->get_char()),
222 false, false);
223 }
224
225 default:
226 throw compiler::unsupported_operation(
227 "Cannot dump operand from type" +
228 std::to_string(value->get_literal_type()));
229 }
