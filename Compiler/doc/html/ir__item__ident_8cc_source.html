22compiler::ir::Operand* compiler::Item_ident::eval_runtime_helper(
23 compiler::ir::IRContext* const ir_context) const {
24 compiler::Symbol_table* const symbol_table = ir_context->get_symbol_table();
25 try {
26 compiler::Symbol_const* const symbol = symbol_table->find_const(name);
27 if (symbol->get_is_pointer()) {
28 throw compiler::fatal_error(
29 "Cannot evaluate pointer / array type. What: " + name);
30 } else {
31 return compiler::ir::dump_value(symbol->get_values()[0]);
32 }
33 } catch (const std::exception& e) {
34 if (opt_level > 0) {
35 compiler::Symbol* const symbol_cur = symbol_table->find_symbol(name);
36 compiler::Symbol_const* symbol_assign = symbol_table->find_assign_const(symbol_cur->get_name());
37 return compiler::ir::dump_value(symbol_assign->get_values()[0]);
38 }
39 throw e;
40 }
41}

43compiler::ir::Operand* compiler::Item_ident::eval_runtime_helper(
44 compiler::ir::IRContext* const ir_context,
45 std::vector<compiler::ir::IR>& ir_list) const {
46 try {
47 compiler::Symbol_table* const symbol_table = ir_context->get_symbol_table();
48 compiler::Symbol_const* const symbol =
49 symbol_table->find_assign_const(name);
50
51 if (symbol->get_is_pointer() || opt_level == 0) {
52 return new compiler::ir::Operand(symbol->get_name());
53 } else {
54 try {
55 return compiler::ir::dump_value(symbol->get_values()[0]);
56 } catch (const std::exception& e) {
57 // Cannot dump value... We return its name.
58 return new compiler::ir::Operand(name);
59 }
60 }
61 } catch (const std::exception& e) {
62 std::cerr << e.what() << std::endl;
63 throw e;
64 }
