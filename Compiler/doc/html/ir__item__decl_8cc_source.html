27  const std::string name = identifier->get_name();

31  // Check current scope.
32  if (ir_context->is_global_context() == true) {

52 void compiler::Item_decl_var_init::generate_ir_helper(
53  ir::IRContext* const ir_context, std::vector<ir::IR>& ir_list,
54  const basic_type& b_type) const {
55  try {
56  const std::string name = identifier->get_name();
57  std::string name_symbol;
58  ir::Operand* const result = expression->eval_runtime(ir_context, ir_list);
59  // Check current scope.
60  if (ir_context->is_global_context()) {
61  name_symbol = ir::global_sign;
62  ir_list.emplace_back(compiler::ir::op_type::GLOBAL_BEGIN, name_symbol);
63  ir_list.emplace_back(compiler::ir::op_type::GLOBAL, result);
64  ir_list.emplace_back(compiler::ir::op_type::GLOBAL_END, name_symbol);
65  } else {
66  const uint32_t scope_id =
67  ir_context->get_symbol_table()->get_top_scope_uuid();
68  name_symbol = ir::local_sign + std::to_string(scope_id);
69  }
70  // Insert into symbol table.
71  compiler::Symbol* symbol = nullptr;
72  if (is_const == true) {
73  symbol = new compiler::Symbol_const(name_symbol,
74  compiler::symbol_type::VAR_TYPE,
75  result->get_value(), false);
76  ir_context->get_symbol_table()->add_const(
77  identifier->get_name(),
78  dynamic_cast<compiler::Symbol_const*>(symbol));
79  } else {
80  symbol = new compiler::Symbol(name_symbol,
81  compiler::symbol_type::VAR_TYPE, false);
82  symbol->set_value(result->get_value());
83  ir_context->get_symbol_table()->add_symbol(identifier->get_name(),
84  symbol);
85  }
86 
87  if (ir_context->is_global_context() == false) {
88 
89  compiler::Item_stmt_assign* const assignment =
90  new compiler::Item_stmt_assign(lineno, identifier, expression);
91  assignment->generate_ir(ir_context, ir_list);
92  }
93  } catch (const std::exception& e) {
94  std::cerr << termcolor::red << termcolor::bold << e.what() << std::endl;
95  exit(1);
96  }

122 uint32_t compiler::Item_decl_array::calculate_array_size(
123  compiler::ir::IRContext* const ir_context,
124  std::vector<compiler::ir::IR>& ir_list, const compiler::basic_type& b_type,
125  std::vector<ir::Operand*>& array_shape) const {
126  // Determine the shape of the array.
127  for (compiler::Item_expr* const shape : identifier->get_array_shape()) {
128  ir::Operand* const value = shape->eval_runtime(ir_context, ir_list);
129  if (value->get_type() == ir::var_type::DB) {
130  throw compiler::fatal_error("Array shape cannot be a double type!");
131  }
132  array_shape.emplace_back(value);
133  }
134 
135  uint32_t array_size = 1;
136  for (ir::Operand* const shape : array_shape) {
137  int size = 0;
138  if ((size = std::stoi(shape->get_value())) < 0) {
139  throw compiler::fatal_error("The array has negative shape!");
140  }
141  array_size *= (uint32_t)size;
142  }
143 
144  return array_size;
145 }

147 void compiler::Item_decl_array::generate_ir_helper(
148  compiler::ir::IRContext* const ir_context,
149  std::vector<compiler::ir::IR>& ir_list,
150  const compiler::basic_type& b_type) const {
151  try {
152  // Determine the size of the array.
153  std::vector<ir::Operand*> array_shape;
154  const uint32_t array_size =
155  calculate_array_size(ir_context, ir_list, b_type, array_shape);
156 
157  ir::Operand* const operand_alloc =
158  new ir::Operand(ir::var_type::NONE,
159  ir::var_type_to_string(compiler::to_ir_type(b_type)) +
160  " * " + std::to_string(array_size),
161  "");
162 
163  // Handle global array type.
164  const std::string name = identifier->get_name();
165  if (ir_context->is_global_context()) {
166  const std::string name_symbol = ir::global_sign + name + "[]";
167  ir_context->get_symbol_table()->add_symbol(
168  name,
169  new compiler::Symbol(name_symbol, compiler::symbol_type::ARRAY_TYPE,
170  false, array_shape));
171  ir_list.emplace_back(ir::op_type::GLOBAL_BEGIN, name_symbol);
172  ir_list.emplace_back(ir::op_type::GLOBAL, operand_alloc);
173  ir_list.emplace_back(
174  ir::op_type::SPACE,
175  new ir::Operand(std::to_string(
176  ir::to_byte_length(compiler::to_ir_type(b_type)) * array_size)));
177  ir_list.emplace_back(ir::op_type::GLOBAL_END, name_symbol);
178  // Handle local array type.
179  } else {
180  const std::string name_symbol =
181  ir::local_sign +
182  std::to_string(ir_context->get_symbol_table()->get_available_id()) +
183  name + "[]";
184  ir_context->get_symbol_table()->add_symbol(
185  name,
186  new compiler::Symbol(name_symbol, compiler::symbol_type::ARRAY_TYPE,
187  true, array_shape));
188  }
189  } catch (const std::exception& e) {
190  std::cerr << termcolor::red << termcolor::bold << lineno << ": " << e.what()
191  << termcolor::reset << std::endl;
192  exit(1);
193  }

245  const std::vector<compiler::Item_literal_array_init*> value_list,

247  compiler::ir::IRContext* const ir_context,
248  std::vector<compiler::ir::IR>& ir_list,
249  const ir::var_type& var_type) const {
250  auto lambda_handle_init = [&](ir::Operand* const array_size,
251  const bool& malloc = false) {
252  // Get the length of the array item.
253  const uint32_t byte_length = ir::to_byte_length(var_type);
254  if (malloc == true) {
255  // Iterate through the size of the array.
256  const uint32_t size = std::stoul(array_size->get_value());
257  for (uint32_t i = 0; i < size; i++) {
258  init_value.emplace_back(
259  new ir::Operand(var_type, "", "0", false, false));
260  }
261  if (ir_context->is_global_context()) {
262  ir_list.emplace_back(
263  ir::op_type::SPACE,
264  new ir::Operand(std::to_string(byte_length * size)));
265  }
266  } else {
267  init_value.emplace_back(new ir::Operand(var_type, "", "0", false, false));
268  if (ir_context->is_global_context()) {
269  ir_list.emplace_back(ir::op_type::SPACE, array_size);
270  } else {
271  const std::string array_name_local =
272  ir_context->get_symbol_table()
273  ->find_symbol(get_identifier()->get_name())
274  ->get_name();
275  const uint32_t ind = byte_length * (init_value.size() - 1);
276  ir_list.emplace_back(
277  ir::op_type::STORE, nullptr, new ir::Operand(array_name_local),
278  new ir::Operand(ir::var_type::i32, "", std::to_string(ind), false,
279  false));
280  }
281  }
282  };
