22void compiler::Item_decl_var::generate_ir_helper(
23 compiler::ir::IRContext* const ir_context,
24 std::vector<compiler::ir::IR>& ir_list, const basic_type& b_type) const {
25 const std::string name = identifier->get_name();
26 // Check current scope.
27 if (ir_context->is_global_context() == true) {
28 const std::string name_symbol = "@" + name;
29 ir_list.emplace_back(compiler::ir::op_type::BEGIN_DATA, name_symbol);
30 ir_list.emplace_back(
31 compiler::ir::op_type::WORD,
32 new compiler::ir::Operand(compiler::to_ir_type(b_type), name, "uninit",
33 true, false));
34 ir_list.emplace_back(compiler::ir::op_type::END_DATA, name_symbol);
35
36 // Create a new symbol for the symbol table.
37 compiler::Symbol* const symbol = new compiler::Symbol(
38 name_symbol, compiler::symbol_type::VAR_TYPE, false);
39 ir_context->get_symbol_table()->add_symbol(name, symbol);
40 } else {
41 const uint32_t scope_id =
42 ir_context->get_symbol_table()->get_top_scope_uuid();
43 const std::string name_symbol = "%" + std::to_string(scope_id) + name;
44 // Create a temporary symbol for the symbol table.
45 compiler::Symbol* const symbol =
46 new compiler::Symbol(name, compiler::symbol_type::VAR_TYPE, false);
47 ir_context->get_symbol_table()->add_symbol(name, symbol);
48 }
49}
