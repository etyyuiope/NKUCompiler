27compiler::ir::Operand* compiler::Item_literal::eval_runtime_helper(
28 compiler::ir::IRContext* const ir_context) const {
29 switch (get_literal_type()) {
30 case literal_type::INT_TYPE: {
31 const auto* const literal =
32 static_cast<const compiler::Item_literal_int* const>(this);
33 return new compiler::ir::Operand(ir::var_type::i32, "",
34 std::to_string(literal->get_int()),
35 false, false);
36 }
37
38 case literal_type::CHAR_TYPE: {
39 const auto* const literal =
40 static_cast<const compiler::Item_literal_char* const>(this);
41 return new compiler::ir::Operand(ir::var_type::i8, "",
42 std::to_string(literal->get_char()),
43 false, false);
44 }
45
46 case literal_type::REAL_TYPE: {
47 const auto* const literal =
48 static_cast<const compiler::Item_literal_real* const>(this);
49 return new compiler::ir::Operand(
50 ir::var_type::DB, "", ir::convert_from_double(literal->get_double()),
51 false, false);
52 }
53
54 default:
55 throw compiler::unimplemented_error(
56 "Cannot do this at this time! Maybe later:)");
57 }
