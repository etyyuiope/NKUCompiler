35std::vector<std::string> compiler::process_input(const std::string& input) {
36 std::vector<std::string> files;
37
38 for (const auto& entry : std::filesystem::directory_iterator(input)) {
39 const std::filesystem::path file_path = entry.path();
40 if (file_path.extension().compare(".sy") == 0) {
41 const std::string sy_name = file_path.string();
42 files.emplace_back(sy_name);
43 }
44 }
45
46 std::filesystem::create_directory(input + "/output");
47
48 std::sort(files.begin(), files.end());
49 return files;
50}

81compiler::Compiler_runtime::Compiler_runtime(const cxxopts::ParseResult& result)
82 : compile_on(result["compile"].as<bool>()),
83 debug_on(result["debug"].as<bool>()),
84 print_ast(result["print-ast"].as<bool>()),
85 print_ir(result["print-ir"].as<bool>()),
86 opt_level(result["opt-level"].as<int>()) {
87 ::opt_level = opt_level;
88 const std::string input = result["source"].as<std::string>();
89
90 if (is_dir(input)) {

92 base_path = input;

94 output_file.open(result["output"].as<std::string>(), std::ios::out);
95 input_file.emplace_back(input);

98 std::cout << "\033[4;90;107mTakanashi Compiler is running!!\033[0m"
99 << std::endl;

102void compiler::Compiler_runtime::run(void) {
103 try {
104 for (uint32_t i = 0; i < input_file.size(); i++) {
105 std::cout << "Reading " << input_file[i] << std::endl;
106 yyset_in(fopen(input_file[i].data(), "r"));
107 yyset_lineno(1);
108 yyparse();
109 yylex_destroy();
110
111 std::string res;
112 std::ostringstream oss;
113 if (!output_file.is_open()) {
114 const std::string file =
115 input_file[i].substr(input_file[i].find_last_of("/"));
116 if (print_ast) {
117 oss << base_path << "/output/" << file << ".ast";
118 output_file.open(oss.str(), std::ios::out);
119 }
120 if (print_ir) {
121 oss << base_path << "/output/" << file << ".ir";
122 output_file.open(oss.str(), std::ios::out);
123 }
124 }
125
126 if (print_ast) {
127 res = root->print_result(0, false);
128 }
129 if (print_ir) {
130 compiler::ir::IRContext* const ir_context =
131 new compiler::ir::IRContext();
132 std::vector<compiler::ir::IR> ir_list;
133 root->generate_ir(ir_context, ir_list);
134
135 for (auto item : ir_list) {
136 item.emit_ir(oss, false);
137 }
138
139 res = oss.str();
140 }
141
142 output_file << res;
143 output_file.flush();
144 output_file.close();
145 }
146
147 } catch (const std::exception& e) {
148 // Error handler.
149 std::cerr << e.what() << std::endl;
150 }
151}
