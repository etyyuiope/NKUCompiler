22compiler::Symbol* compiler::Symbol_table::find_symbol(const std::string& name)
23{
24 // Iterate over the current table and the parent table.
25 last_uuid = symbol_table.size() - 1;
26 Symbol* result = nullptr;
27 for (auto& cur : symbol_table) {
28 result = cur->find_symbol(name);
29 if (result != nullptr) {
30 break;
31 }
32 last_uuid--;
33 }
34 if (result != nullptr) {
35 return result;
36 }
37
38 // Not found. Raise an error.
39 std::ostringstream oss;
40 oss << "Error: Symbol " << name << " is not declaraed.";
41 throw compiler::undeclared_symbol(oss.str());
42 return nullptr;
43}

54compiler::Symbol_block* compiler::Symbol_table::get_spec_block(const uint32_t& i)
55{
56 uint32_t index = symbol_table.size() - i;
57 if (index <= 0) {
58 throw compiler::fatal_error("Error: The global symbol table is not found!");
59 }
60 auto iter = symbol_table.begin();
61 for (uint32_t i = 0; i < index; i++) {
62 iter++;
63 }
64 return *iter;
65}
