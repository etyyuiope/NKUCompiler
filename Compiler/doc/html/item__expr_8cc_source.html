51std::string compiler::Item_expr_comma::print_result(const uint32_t& indent,
52 const bool& leaf) const {
53 std::ostringstream oss;

55 oss << " Comma Expression" << std::endl;
56 for (uint32_t i = 0; i < expressions.size(); i++) {
