107 compiler::ir::Operand* compiler::Item_expr_binary::eval_runtime_helper(
108  compiler::ir::IRContext* const ir_context) const {
109  // TODO:: Type check + new node creation.
110  // There could multiple situations:
111  // 1. numeric + numeric;
112  // 2. numeric + pointer;
113  // 3. pointer + pointer;
114  // 4. Array cannot be added to any other types, unless it is access through
115  // indexing.
116  compiler::ir::Operand* const expr_left = lhs->eval_runtime(ir_context);
117  compiler::ir::Operand* const expr_right = rhs->eval_runtime(ir_context);
118  // Check type compatibility.
119 
120  if (!compiler::ir::check_valid_binary(expr_left, expr_right)) {
121  throw compiler::unsupported_operation(
122  "Error: Operand types are incompatible!");
123  }
124 
125  switch (type) {
126  case compiler::binary_type::ADD_TYPE: {
127  const double lhs =
128  compiler::ir::convert_from_string(expr_left->get_value());
129  const double rhs =
130  compiler::ir::convert_from_string(expr_right->get_value());
131  const double res = lhs + rhs;
132  const compiler::ir::var_type type =
133  compiler::ir::get_type_priority(expr_left->get_type(),
134  expr_right->get_type())
135  ? expr_left->get_type()
136  : expr_right->get_type();
137  return new compiler::ir::Operand(
138  type, "", compiler::ir::convert_from_double(res), false, false);
139  }
140 
141  // TODO: Implement other types. Maybe we can add a utility function?
142  default:
143  throw compiler::unsupported_operation("Unknown binary type!");
144  }
145 }

148 compiler::ir::Operand* compiler::Item_expr_binary::eval_runtime_helper(
149  compiler::ir::IRContext* const ir_context,
150  std::vector<compiler::ir::IR>& ir_list) const {
151  // First check the optimization level. If so, we calculate the expression at
152  // runtime.
153  if (opt_level > 0) {
154  try {
155  return eval_runtime_helper(ir_context);
156  } catch (const std::exception& e) {
157  std::cerr << termcolor::red << termcolor::bold << e.what()
158  << termcolor::reset << std::endl;
159  exit(1);
160  }
161  // Generate IR.
162  } else {
163  ir::Operand* const dst = new ir::Operand(
164  ir::local_sign +
165  std::to_string(ir_context->get_symbol_table()->get_available_id()));
166  ir::Operand* left = nullptr;
167  ir::Operand* right = nullptr;
168 
169  if (type != LAND_TYPE && type != LOR_TYPE) {
170  if (opt_level > 0) {
171  left = lhs->eval_runtime(ir_context);
172  right = rhs->eval_runtime(ir_context);
173  } else {
174  left = lhs->eval_runtime(ir_context, ir_list);
175  right = rhs->eval_runtime(ir_context, ir_list);
176  }
177  }
178 
179  switch (type) {
180  // TODO: Implement implicit type convertion.
181  case ADD_TYPE: {
182  ir_list.emplace_back(ir::op_type::IADD, dst, left, right);
183  break;
184  }
185  case SUB_TYPE: {
186  ir_list.emplace_back(ir::op_type::ISUB, dst, left, right);
187  break;
188  }
189  case MUL_TYPE: {
190  // Can do optimization.
191  ir_list.emplace_back(ir::op_type::IMUL, dst, left, right);
192  break;
193  }
194  case DIV_TYPE: {
195  // Can do optimization.
196  ir_list.emplace_back(ir::op_type::IDIV, dst, left, right);
197  break;
198  }
199  case MOD_TYPE: {
200  // a mod b => a - ceil(a / b) * b.
201  ir_list.emplace_back(ir::op_type::IMOD, dst, left, right);
202  break;
203  }
204  case NEQ_TYPE: {
205  ir_list.emplace_back(ir::op_type::CMP, nullptr, left, right);
206  ir_list.emplace_back(ir::op_type::MOVNE, dst, OPERAND_VALUE("1"),
207  OPERAND_VALUE("1"));
208  break;
209  }
210  case EQ_TYPE: {
211  ir_list.emplace_back(ir::op_type::CMP, nullptr, left, right);
212  ir_list.emplace_back(ir::op_type::MOVEQ, dst, OPERAND_VALUE("1"),
213  OPERAND_VALUE("1"));
214  break;
215  }
216  case G_TYPE: {
217  ir_list.emplace_back(ir::op_type::CMP, nullptr, left, right);
218  ir_list.emplace_back(ir::op_type::MOVGT, dst, OPERAND_VALUE("1"),
219  OPERAND_VALUE("1"));
220  break;
221  }
222  case L_TYPE: {
223  ir_list.emplace_back(ir::op_type::CMP, nullptr, left, right);
224  ir_list.emplace_back(ir::op_type::MOVLT, dst, OPERAND_VALUE("1"),
225  OPERAND_VALUE("1"));
226  break;
227  }
228  case LE_TYPE: {
229  ir_list.emplace_back(ir::op_type::CMP, nullptr, left, right);
230  ir_list.emplace_back(ir::op_type::MOVLE, dst, OPERAND_VALUE("1"),
231  OPERAND_VALUE("1"));
232  break;
233  }
234  case GE_TYPE: {
235  ir_list.emplace_back(ir::op_type::CMP, nullptr, left, right);
236  ir_list.emplace_back(ir::op_type::MOVGE, dst, OPERAND_VALUE("1"),
237  OPERAND_VALUE("1"));
238  break;
239  }
240  case BITAND_TYPE: {
241  ir_list.emplace_back(ir::op_type::BAND, dst, left, right);
242  break;
243  }
244  case BITOR_TYPE: {
245  ir_list.emplace_back(ir::op_type::BOR, dst, left, right);
246  break;
247  }
248  case BITXOR_TYPE: {
249  ir_list.emplace_back(ir::op_type::BXOR, dst, left, right);
250  break;
251  }
252 
253  default:
254  throw compiler::unsupported_operation(
255  "Error: Unrecognized binary type!");
256  }
257  return dst;
258  }

261 compiler::ir::Operand* compiler::Item_expr_unary::eval_runtime_helper(
262  compiler::ir::IRContext* const ir_context,
263  std::vector<compiler::ir::IR>& ir_list) const {
264  if (opt_level > 0) {
265  return eval_runtime_helper(ir_context);
266  } else {
267  // Create a dst operand.
268  ir::Operand* const dst = new ir::Operand(
269  ir::local_sign +
270  std::to_string(ir_context->get_symbol_table()->get_available_id()));
271 
272  switch (type) {
273  case LNOT_TYPE: {
274  ir_list.emplace_back(ir::op_type::CMP, nullptr, OPERAND_VALUE("1"),
275  expr->eval_runtime(ir_context, ir_list));
276  ir_list.emplace_back(ir::op_type::MOVNE, dst, OPERAND_VALUE("1"),
277  OPERAND_VALUE("1"));
278  break;
279  }
280  case BITNEG_TYPE: {
281  ir_list.emplace_back(ir::op_type::BNEG, dst,
282  expr->eval_runtime(ir_context, ir_list));
283  break;
284  }
285  case UMINUS_TYPE: {
286  ir_list.emplace_back(ir::op_type::ISUB, dst, OPERAND_VALUE("1"),
287  expr->eval_runtime(ir_context, ir_list));
288  break;
289  }
290  case UADD_TYPE: {
291  // No need to generate any IR for it.
292  break;
293  }
294  default: {
295  throw compiler::fatal_error("Error: Unknown Unary expression type!");
296  }
297  }
298  return dst;
299  }
