52 stack.push_back(

98compiler::ir::Operand* compiler::Item_expr_binary::eval_runtime_helper(
99 compiler::ir::IRContext* const ir_context) const {
100 // TODO:: Type check + new node creation.
101 // There could multiple situations:
102 // 1. numeric + numeric;
103 // 2. numeric + pointer;
104 // 3. pointer + pointer;
105 // 4. Array cannot be added to any other types.
106 compiler::ir::Operand* const expr_left = lhs->eval_runtime(ir_context);
107 compiler::ir::Operand* const expr_right = rhs->eval_runtime(ir_context);
108 // Check type compatibility.
109
110 if (!compiler::ir::check_valid_binary(expr_left, expr_right)) {
111 throw compiler::unsupported_operation("Type incompatible!");
112 }
113
114 switch (type) {
115 case compiler::binary_type::ADD_TYPE: {
116 const double lhs =
117 compiler::ir::convert_from_string(expr_left->get_value());
118 const double rhs =
119 compiler::ir::convert_from_string(expr_right->get_value());
120 const double res = lhs + rhs;
121 const compiler::ir::var_type type =
122 compiler::ir::get_type_priority(expr_left->get_type(),
123 expr_right->get_type())
124 ? expr_left->get_type()
125 : expr_right->get_type();
126 return new compiler::ir::Operand(
127 type, "", compiler::ir::convert_from_double(res), false, false);
128 }
129
130 // TODO: Implement other types. Maybe we can add a utility function?
131 default:
132 throw compiler::unsupported_operation("Unknown binary type!");
133 }
134}

140{
141 if (opt_level > 0) {
142 auto cond_res = eval_runtime(ir_context);

148 return std::make_pair(ir::op_type::NOP, ir::op_type::JMP);
