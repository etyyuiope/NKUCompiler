102 compiler::ir::Operand* compiler::Item_expr_binary::eval_runtime_helper(
103  compiler::ir::IRContext* const ir_context) const {
104  // TODO:: Type check + new node creation.
105  // There could multiple situations:
106  // 1. numeric + numeric;
107  // 2. numeric + pointer;
108  // 3. pointer + pointer;
109  // 4. Array cannot be added to any other types.
110  compiler::ir::Operand* const expr_left = lhs->eval_runtime(ir_context);
111  compiler::ir::Operand* const expr_right = rhs->eval_runtime(ir_context);
112  // Check type compatibility.
113 
114  if (!compiler::ir::check_valid_binary(expr_left, expr_right)) {
115  throw compiler::unsupported_operation("Type incompatible!");
116  }
117 
118  switch (type) {
119  case compiler::binary_type::ADD_TYPE: {
120  const double lhs =
121  compiler::ir::convert_from_string(expr_left->get_value());
122  const double rhs =
123  compiler::ir::convert_from_string(expr_right->get_value());
124  const double res = lhs + rhs;
125  const compiler::ir::var_type type =
126  compiler::ir::get_type_priority(expr_left->get_type(),
127  expr_right->get_type())
128  ? expr_left->get_type()
129  : expr_right->get_type();
130  return new compiler::ir::Operand(
131  type, "", compiler::ir::convert_from_double(res), false, false);
132  }
133 
134  // TODO: Implement other types. Maybe we can add a utility function?
135  default:
136  throw compiler::unsupported_operation("Unknown binary type!");
137  }
138 }
