466  std::string short_sw, long_sw;
467  const char* pdata = text.c_str();
468  if (isalnum(*pdata) && *(pdata + 1) == ',') {

494  if (strncmp(pdata, "--", 2) == 0) {

659 template <typename T>
660 void integer_parser(const std::string& text, T& value) {
661  parser_tool::IntegerDesc int_desc = parser_tool::SplitInteger(text);
662 
663  using US = typename std::make_unsigned<T>::type;
664  constexpr bool is_signed = std::numeric_limits<T>::is_signed;
665 
666  const bool negative = int_desc.negative.length() > 0;
667  const uint8_t base = int_desc.base.length() > 0 ? 16 : 10;
668  const std::string& value_match = int_desc.value;
669 
670  US result = 0;
671 
672  for (char ch : value_match) {
673  US digit = 0;
674 
675  if (ch >= '0' && ch <= '9') {
676  digit = static_cast<US>(ch - '0');
677  } else if (base == 16 && ch >= 'a' && ch <= 'f') {
678  digit = static_cast<US>(ch - 'a' + 10);
679  } else if (base == 16 && ch >= 'A' && ch <= 'F') {
680  digit = static_cast<US>(ch - 'A' + 10);
681  } else {
682  throw_or_mimic<argument_incorrect_type>(text);
683  }
684 
685  const US next = static_cast<US>(result * base + digit);
686  if (result > next) {
687  throw_or_mimic<argument_incorrect_type>(text);
688  }
689 
690  result = next;
691  }
692 
693  detail::check_signed_range<T>(negative, result, text);
694 
695  if (negative) {
696  checked_negate<T>(value, result, text,
697  std::integral_constant<bool, is_signed>());
698  } else {
699  value = static_cast<T>(result);
700  }
701 }

1350  size_t allowed, bool tab_expansion) {
1351  auto desc = o.desc;
1352 
1353  if (o.has_default && (!o.is_boolean || o.default_value != "false")) {
1354  if (!o.default_value.empty()) {
1355  desc += toLocalString(" (default: " + o.default_value + ")");
1356  } else {
1357  desc += toLocalString(" (default: \"\")");
1358  }
1359  }
1360 
1361  String result;
1362 
1363  if (tab_expansion) {
1364  String desc2;
1365  auto size = size_t{0};
1366  for (auto c = std::begin(desc); c != std::end(desc); ++c) {
1367  if (*c == '\n') {
1368  desc2 += *c;
1369  size = 0;
1370  } else if (*c == '\t') {
1371  auto skip = 8 - size % 8;
1372  stringAppend(desc2, skip, ' ');
1373  size += skip;
1374  } else {
1375  desc2 += *c;
1376  ++size;
1377  }
1378  }
1379  desc = desc2;
1380  }
1381 
1382  desc += " ";
1383 
1384  auto current = std::begin(desc);
1385  auto previous = current;
1386  auto startLine = current;
1387  auto lastSpace = current;
1388 
1389  auto size = size_t{};
1390 
1391  bool appendNewLine;
1392  bool onlyWhiteSpace = true;
1393 
1394  while (current != std::end(desc)) {
1395  appendNewLine = false;
1396 
1397  if (std::isblank(*previous)) {
1398  lastSpace = current;
1399  }
1400 
1401  if (!std::isblank(*current)) {
1402  onlyWhiteSpace = false;
1403  }
1404 
1405  while (*current == '\n') {
1406  previous = current;
1407  ++current;
1408  appendNewLine = true;
1409  }
1410 
1411  if (!appendNewLine && size >= allowed) {
1412  if (lastSpace != startLine) {
1413  current = lastSpace;
1414  previous = current;
1415  }
1416  appendNewLine = true;

1589

1591  if (strcmp(argv[current], "--") == 0) {
1592  consume_remaining = true;

1597  values::parser_tool::ArguDesc argu_desc =
1598  values::parser_tool::ParseArgument(argv[current], matched);
1599

1601  // not a flag
1602 
1603  // but if it starts with a `-`, then it's an error
1604  if (argv[current][0] == '-' && argv[current][1] != '\0') {
1605  if (!m_allow_unrecognised) {
1606  throw_or_mimic<option_syntax_exception>(argv[current]);
1607  }
1608  }
1609 
1610  // if true is returned here then it was consumed, otherwise it is
1611  // ignored
1612  if (consume_positional(argv[current], next_positional)) {
1613  } else {
1614  unmatched.emplace_back(argv[current]);
1615  }
1616  // if we return from here then it was parsed successfully, so continue
1617  } else {
1618  // short or long option?
1619  if (argu_desc.grouping) {
1620  const std::string& s = argu_desc.arg_name;
1621

1623  std::string name(1, s[i]);

1625

1627  if (m_allow_unrecognised) {
1628  unmatched.emplace_back(std::string("-") + s[i]);
1629  continue;
1630  }
1631  // error
1632  throw_or_mimic<option_not_exists_exception>(name);
1633  }
1634

1636 
1637  if (i + 1 == s.size()) {
1638  // it must be the last argument
1639  checked_parse_arg(argc, argv, current, value, name);
1640  } else if (value->value().has_implicit()) {
1641  parse_option(value, name, value->value().get_implicit_value());
1642  } else if (i + 1 < s.size()) {
1643  std::string arg_value = s.substr(i + 1);
1644  parse_option(value, name, arg_value);
1645  break;
1646  } else {
1647  // error
1648  throw_or_mimic<option_requires_argument_exception>(name);
1649  }
1650  }
1651  } else if (argu_desc.arg_name.length() != 0) {
1652  const std::string& name = argu_desc.arg_name;
1653 
1654  auto iter = m_options.find(name);
1655 
1656  if (iter == m_options.end()) {
1657  if (m_allow_unrecognised) {
1658  // keep unrecognised options in argument list, skip to next argument
1659  unmatched.emplace_back(argv[current]);
1660  ++current;
1661  continue;
1662  }
1663  // error
1664  throw_or_mimic<option_not_exists_exception>(name);
1665  }
1666 
1667  auto opt = iter->second;
1668 
1669  // equals provided for long option?
1670  if (argu_desc.set_value) {
1671  // parse the option given
1672 
1673  parse_option(opt, name, argu_desc.value);
1674  } else {
1675  // parse the next argument
1676  checked_parse_arg(argc, argv, current, opt, name);
1677  }
1678  }
1679  }
1680 
1681  ++current;
1682  }
1683 
1684  for (auto& opt : m_options) {
1685  auto& detail = opt.second;
1686  const auto& value = detail->value();
1687 
1688  auto& store = m_parsed[detail->hash()];

1696  }
1697  }

1703  }
1704  ++current;
1705  }

1711  }
1712  }
