384template <typename T>
385void throw_or_mimic(const std::string& text) {
386 static_assert(std::is_base_of<std::exception, T>::value,
387 "throw_or_mimic only works on std::exception and "
388 "deriving classes");
389
390#ifndef CXXOPTS_NO_EXCEPTIONS
391 // If CXXOPTS_NO_EXCEPTIONS is not defined, just throw
392 throw T{text};
393#else
394 // Otherwise manually instantiate the exception, print what() to stderr,
395 // and exit
396 T exception{text};
397 std::cerr << exception.what() << std::endl;
398 std::exit(EXIT_FAILURE);
399#endif
400}

464inline std::pair<std::string, std::string> SplitSwitchDef(
465 const std::string& text) {
466 std::string short_sw, long_sw;
467 const char* pdata = text.c_str();
468 if (isalnum(*pdata) && *(pdata + 1) == ',') {
469 short_sw = std::string(1, *pdata);
470 pdata += 2;
471 }
472 while (*pdata == ' ') {
473 pdata += 1;
474 }
475 if (isalnum(*pdata)) {
476 const char* store = pdata;
477 pdata += 1;
478 while (isalnum(*pdata) || *pdata == '-' || *pdata == '_') {
479 pdata += 1;
480 }
481 if (*pdata == '\0') {
482 long_sw = std::string(store, pdata - store);
483 } else {
484 throw_or_mimic<invalid_option_format_error>(text);
485 }
486 }
487 return std::pair<std::string, std::string>(short_sw, long_sw);
488}

490inline ArguDesc ParseArgument(const char* arg, bool& matched) {
491 ArguDesc argu_desc;
492 const char* pdata = arg;
493 matched = false;
494 if (strncmp(pdata, "--", 2) == 0) {
495 pdata += 2;
496 if (isalnum(*pdata)) {
497 argu_desc.arg_name.emplace_back(*pdata);
498 pdata += 1;
499 while (isalnum(*pdata) || *pdata == '-' || *pdata == '_') {
500 argu_desc.arg_name.emplace_back(*pdata);
501 pdata += 1;
502 }
503 if (argu_desc.arg_name.length() > 1) {
504 if (*pdata == '=') {
505 argu_desc.set_value = true;
506 pdata += 1;
507 if (*pdata != '\0') {
508 argu_desc.value = std::string(pdata);
509 }
510 matched = true;
511 } else if (*pdata == '\0') {
512 matched = true;
513 }
514 }
515 }
516 } else if (strncmp(pdata, "-", 1) == 0) {
517 pdata += 1;
518 argu_desc.grouping = true;
519 while (isalnum(*pdata)) {
520 argu_desc.arg_name.emplace_back(*pdata);
521 pdata += 1;
522 }
523 matched = !argu_desc.arg_name.empty() && *pdata == '\0';
524 }
525 return argu_desc;
526}

543inline IntegerDesc SplitInteger(const std::string& text) {
544 std::smatch match;
545 std::regex_match(text, match, integer_pattern);
546
547 if (match.length() == 0) {
548 throw_or_mimic<argument_incorrect_type>(text);
549 }
550
551 IntegerDesc desc;
552 desc.negative = match[1];
553 desc.base = match[2];
554 desc.value = match[3];
555
556 if (match.length(4) > 0) {
557 desc.base = match[5];
558 desc.value = "0";
559 return desc;
560 }
561
562 return desc;
563}

592 std::match_results<const char*> result;
593 std::regex_match(arg, result, option_matcher);
594 matched = !result.empty();

646template <typename R, typename T>
647void checked_negate(R& r, T&& t, const std::string&, std::true_type) {
648 // if we got to here, then `t` is a positive number that fits into
649 // `R`. So to avoid MSVC C4146, we first cast it to `R`.
650 // See https://github.com/jarro2783/cxxopts/issues/62 for more details.
651 r = static_cast<R>(-static_cast<R>(t - 1) - 1);
652}

659template <typename T>
660void integer_parser(const std::string& text, T& value) {
661 parser_tool::IntegerDesc int_desc = parser_tool::SplitInteger(text);
662
663 using US = typename std::make_unsigned<T>::type;
664 constexpr bool is_signed = std::numeric_limits<T>::is_signed;
665
666 const bool negative = int_desc.negative.length() > 0;
667 const uint8_t base = int_desc.base.length() > 0 ? 16 : 10;
668 const std::string& value_match = int_desc.value;
669
670 US result = 0;
671
672 for (char ch : value_match) {
673 US digit = 0;
674
675 if (ch >= '0' && ch <= '9') {
676 digit = static_cast<US>(ch - '0');
677 } else if (base == 16 && ch >= 'a' && ch <= 'f') {
678 digit = static_cast<US>(ch - 'a' + 10);
679 } else if (base == 16 && ch >= 'A' && ch <= 'F') {
680 digit = static_cast<US>(ch - 'A' + 10);
681 } else {
682 throw_or_mimic<argument_incorrect_type>(text);
683 }
684
685 const US next = static_cast<US>(result * base + digit);
686 if (result > next) {
687 throw_or_mimic<argument_incorrect_type>(text);
688 }
689
690 result = next;
691 }
692
693 detail::check_signed_range<T>(negative, result, text);
694
695 if (negative) {
696 checked_negate<T>(value, result, text,
697 std::integral_constant<bool, is_signed>());
698 } else {
699 value = static_cast<T>(result);
700 }
701}

745template <typename T>
746void parse_value(const std::string& text, std::vector<T>& value) {
747 if (text.empty()) {
748 T v;
749 parse_value(text, v);
750 value.emplace_back(std::move(v));
751 return;
752 }
753 std::stringstream in(text);
754 std::string token;
755 while (!in.eof() && std::getline(in, token, CXXOPTS_VECTOR_DELIMITER)) {

1319 const auto& l = o.l;

1321 String result = " ";
1322
1323 if (!s.empty()) {
1324 result += "-" + toLocalString(s);
1325 if (!l.empty()) {
1326 result += ",";
1327 }
1328 } else {
1329 result += " ";
1330 }
1331
1332 if (!l.empty()) {
1333 result += " --" + toLocalString(l);
1334 }
1335
1336 auto arg = !o.arg_help.empty() ? toLocalString(o.arg_help) : "arg";
1337
1338 if (!o.is_boolean) {
1339 if (o.has_implicit) {
1340 result += " [=" + arg + "(=" + toLocalString(o.implicit_value) + ")]";

1348
1349String format_description(const HelpOptionDetails& o, size_t start,
1350 size_t allowed, bool tab_expansion) {
1351 auto desc = o.desc;
1352
1353 if (o.has_default && (!o.is_boolean || o.default_value != "false")) {
1354 if (!o.default_value.empty()) {
1355 desc += toLocalString(" (default: " + o.default_value + ")");
1356 } else {
1357 desc += toLocalString(" (default: \"\")");
1358 }
1359 }
1360
1361 String result;
1362
1363 if (tab_expansion) {
1364 String desc2;
1365 auto size = size_t{0};
1366 for (auto c = std::begin(desc); c != std::end(desc); ++c) {
1367 if (*c == '\n') {
1368 desc2 += *c;
1369 size = 0;
1370 } else if (*c == '\t') {
1371 auto skip = 8 - size % 8;
1372 stringAppend(desc2, skip, ' ');
1373 size += skip;
1374 } else {
1375 desc2 += *c;
1376 ++size;
1377 }
1378 }
1379 desc = desc2;
1380 }
1381
1382 desc += " ";
1383
1384 auto current = std::begin(desc);
1385 auto previous = current;
1386 auto startLine = current;
1387 auto lastSpace = current;
1388
1389 auto size = size_t{};
1390
1391 bool appendNewLine;
1392 bool onlyWhiteSpace = true;
1393
1394 while (current != std::end(desc)) {
1395 appendNewLine = false;
1396
1397 if (std::isblank(*previous)) {
1398 lastSpace = current;
1399 }
1400
1401 if (!std::isblank(*current)) {
1402 onlyWhiteSpace = false;
1403 }
1404
1405 while (*current == '\n') {
1406 previous = current;
1407 ++current;
1408 appendNewLine = true;
1409 }
1410
1411 if (!appendNewLine && size >= allowed) {
1412 if (lastSpace != startLine) {
1413 current = lastSpace;
1414 previous = current;
1415 }
1416 appendNewLine = true;
1417 }
1418
1419 if (appendNewLine) {
1420 stringAppend(result, startLine, current);
1421 startLine = current;
1422 lastSpace = current;
1423
1424 if (*previous != '\n') {
1425 stringAppend(result, "\n");
1426 }
1427
1428 stringAppend(result, start, ' ');
1429
1430 if (*previous != '\n') {
1431 stringAppend(result, lastSpace, current);
1432 }
1433
1434 onlyWhiteSpace = true;
1435 size = 0;
1436 }
1437
1438 previous = current;
1439 ++current;
1440 ++size;
1441 }
1442
1443 // append whatever is left but ignore whitespace
1444 if (!onlyWhiteSpace) {
1445 stringAppend(result, startLine, previous);
1446 }
1447
1448 return result;
1449}

1583inline ParseResult OptionParser::parse(int argc, const char* const* argv) {
1584 int current = 1;
1585 bool consume_remaining = false;
1586 auto next_positional = m_positional.begin();
1587
1588 std::vector<std::string> unmatched;
1589
1590 while (current != argc) {
1591 if (strcmp(argv[current], "--") == 0) {
1592 consume_remaining = true;
1593 ++current;
1594 break;
1595 }
1596 bool matched = false;
1597 values::parser_tool::ArguDesc argu_desc =
1598 values::parser_tool::ParseArgument(argv[current], matched);
1599
1600 if (!matched) {
1601 // not a flag
1602
1603 // but if it starts with a `-`, then it's an error
1604 if (argv[current][0] == '-' && argv[current][1] != '\0') {
1605 if (!m_allow_unrecognised) {
1606 throw_or_mimic<option_syntax_exception>(argv[current]);
1607 }
1608 }
1609
1610 // if true is returned here then it was consumed, otherwise it is
1611 // ignored
1612 if (consume_positional(argv[current], next_positional)) {
1613 } else {
1614 unmatched.emplace_back(argv[current]);
1615 }
1616 // if we return from here then it was parsed successfully, so continue
1617 } else {
1618 // short or long option?
1619 if (argu_desc.grouping) {
1620 const std::string& s = argu_desc.arg_name;
1621
1622 for (std::size_t i = 0; i != s.size(); ++i) {
1623 std::string name(1, s[i]);
1624 auto iter = m_options.find(name);
1625
1626 if (iter == m_options.end()) {
1627 if (m_allow_unrecognised) {
1628 unmatched.emplace_back(std::string("-") + s[i]);
1629 continue;
1630 }
1631 // error
1632 throw_or_mimic<option_not_exists_exception>(name);
1633 }
1634
1635 auto value = iter->second;
1636
1637 if (i + 1 == s.size()) {
1638 // it must be the last argument
1639 checked_parse_arg(argc, argv, current, value, name);
1640 } else if (value->value().has_implicit()) {
1641 parse_option(value, name, value->value().get_implicit_value());
1642 } else if (i + 1 < s.size()) {
1643 std::string arg_value = s.substr(i + 1);
1644 parse_option(value, name, arg_value);
1645 break;
1646 } else {
1647 // error
1648 throw_or_mimic<option_requires_argument_exception>(name);
1649 }
1650 }
1651 } else if (argu_desc.arg_name.length() != 0) {
1652 const std::string& name = argu_desc.arg_name;
1653
1654 auto iter = m_options.find(name);
1655
1656 if (iter == m_options.end()) {
1657 if (m_allow_unrecognised) {
1658 // keep unrecognised options in argument list, skip to next argument
1659 unmatched.emplace_back(argv[current]);
1660 ++current;
1661 continue;
1662 }
1663 // error
1664 throw_or_mimic<option_not_exists_exception>(name);
1665 }
1666
1667 auto opt = iter->second;
1668
1669 // equals provided for long option?
1670 if (argu_desc.set_value) {
1671 // parse the option given
1672
1673 parse_option(opt, name, argu_desc.value);
1674 } else {
1675 // parse the next argument
1676 checked_parse_arg(argc, argv, current, opt, name);
1677 }
1678 }
1679 }
1680
1681 ++current;
1682 }
1683
1684 for (auto& opt : m_options) {
1685 auto& detail = opt.second;
1686 const auto& value = detail->value();
1687
1688 auto& store = m_parsed[detail->hash()];
1689
1690 if (value.has_default()) {
1691 if (!store.count() && !store.has_default()) {
1692 parse_default(detail);
1693 }
1694 } else {
1695 parse_no_value(detail);
1696 }
1697 }
1698
1699 if (consume_remaining) {
1700 while (current < argc) {
1701 if (!consume_positional(argv[current], next_positional)) {
1702 break;
1703 }
1704 ++current;
1705 }
1706
1707 // adjust argv for any that couldn't be swallowed
1708 while (current != argc) {
1709 unmatched.emplace_back(argv[current]);
1710 ++current;
1711 }
1712 }
1713
1714 finalise_aliases();
1715
1716 ParseResult parsed(std::move(m_keys), std::move(m_parsed),
1717 std::move(m_sequential), std::move(unmatched));
1718 return parsed;
1719}

1775inline String Options::help_one_group(const std::string& g) const {
1776 using OptionHelp = std::vector<std::pair<String, String>>;
1777
1778 auto group = m_help.find(g);
1779 if (group == m_help.end()) {
1780 return "";
1781 }
1782
1783 OptionHelp format;
1784
1785 size_t longest = 0;
1786
1787 String result;
1788
1789 if (!g.empty()) {
1790 result += toLocalString(" " + g + " options:\n");
1791 }
1792
1793 for (const auto& o : group->second.options) {
1794 if (m_positional_set.find(o.l) != m_positional_set.end() &&
1795 !m_show_positional) {
1796 continue;
1797 }
1798
1799 auto s = format_option(o);
1800 longest = (std::max)(longest, stringLength(s));
1801 format.emplace_back(std::make_pair(s, String()));
1802 }
1803 longest = (std::min)(longest, OPTION_LONGEST);
1804
1805 // widest allowed description -- min 10 chars for helptext/line
1806 size_t allowed = 10;
1807 if (m_width > allowed + longest + OPTION_DESC_GAP) {
1808 allowed = m_width - longest - OPTION_DESC_GAP;
1809 }
1810
1811 auto fiter = format.begin();
1812 for (const auto& o : group->second.options) {
1813 if (m_positional_set.find(o.l) != m_positional_set.end() &&
1814 !m_show_positional) {
1815 continue;
1816 }
1817
1818 auto d = format_description(o, longest + OPTION_DESC_GAP, allowed,
1819 m_tab_expansion);
1820
1821 result += fiter->first;
1822 if (stringLength(fiter->first) > longest) {
1823 result += '\n';
1824 result += toLocalString(std::string(longest + OPTION_DESC_GAP, ' '));
1825 } else {
1826 result += toLocalString(std::string(
1827 longest + OPTION_DESC_GAP - stringLength(fiter->first), ' '));
1828 }
1829 result += d;
1830 result += '\n';
1831
1832 ++fiter;
1833 }
1834
1835 return result;
1836}
