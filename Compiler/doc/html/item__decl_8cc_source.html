104std::string
105compiler::Item_stmt_decl::print_result(const uint32_t& indent, const bool& leaf) const
106{
107 std::ostringstream oss;
108 print_indent(indent, leaf, oss);
109 oss << " Declaration" << std::endl;
110
111 for (uint32_t i = 0; i < declarations.size(); i++) {
112 oss << declarations[i]->print_result(indent + 2, i == declarations.size() - 1);
113 }
114 return oss.str();
115}
