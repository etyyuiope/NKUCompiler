40void Symbol_table::put(Item_ident* const item)
41{
42 if (symbol_table.size() <= 0) {

49Item_ident*
50Symbol_table::get(const std::string& name)
51{
52 if (symbol_table.size() <= 0) {
53 throw std::runtime_error("There is no symbol table at all!\n");
54 } else {
55 Item_ident* ans = nullptr;
56
57 for (auto item : symbol_table) {
58 if (item.count(name) != 0) {
59 ans = item[name];
60 }
61 }
62
63 if (ans != nullptr) {
64 return ans;
65 } else {
66 throw std::runtime_error("Undeclared symbol!\n");
67 }
68 }
69}
