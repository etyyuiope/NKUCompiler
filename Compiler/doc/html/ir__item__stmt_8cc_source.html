48void compiler::Item_stmt_eif::generate_ir(
49 compiler::ir::IRContext* const ir_context,
50 std::vector<compiler::ir::IR>& ir_list)
51 const
52{
53 try {
54 ir_context->enter_scope();
55 // Get the id of current scope.
56 const uint32_t scope_uuid_cur = (uint32_t)(ir_context->get_symbol_table().get_top_scope_uuid());
57 const BranchIR branch_ir = condition->eval_cond(ir_context, ir_list);
58
59 // The if_branch is taken. Evaluated at runtime.
60 if (branch_ir.first == compiler::ir::JMP) {
61 if_branch->generate_ir(ir_context, ir_list);
62 return;
63 } else if (branch_ir.second == compiler::ir::JMP) {
64 else_branch->generate_ir(ir_context, ir_list);
65 return;
66 }
67
68 // If branches cannot be evaluated at runtime, we should create IRs for each branch.
69 std::ostringstream oss;
70 oss << ".L." << scope_uuid_cur << "_eif";
71 ir_list.emplace_back(branch_ir.second, oss.str());
72 oss.str("");
73 oss.flush();
74
75 std::vector<compiler::ir::IR> ir_if, ir_else;
76 compiler::ir::IRContext ir_context_if, ir_context_else;
77
78 ir_context_if.enter_scope();
79 if_branch->generate_ir(&ir_context_if, ir_if);
80 ir_context_if.leave_scope();
81
82 ir_context_else.enter_scope();
83 else_branch->generate_ir(&ir_context_else, ir_else);
84 ir_context_else.leave_scope();
85
86 std::vector<compiler::ir::IR> ir_end;
87 oss << ".L." << scope_uuid_cur << "_endif";
88 ir_list.emplace_back(branch_ir.second, oss.str());
89 ir_end.emplace_back(compiler::ir::op_type::LBL, oss.str());
90
91 // The actual symbol table is wrapped twice.
92 const auto& symbol_table = ir_context_if.get_symbol_table().get_symbol_table();
93 // Traverse the symbol table.
94 for (auto iter = symbol_table.cend(); iter != symbol_table.cbegin(); iter--) {
95 const compiler::symbol_table_type cur_table = (*iter)->get_block();
96
97 for (auto& entry : cur_table) {
98 const std::string name = entry.second->get_name();
99 // TODO: FIND VARIABLE.
100 }
101 }
102
103 ir_context->leave_scope();
104 } catch (...) {
105 // FIXME: Determine which exception should be caught.
106 }
107}
