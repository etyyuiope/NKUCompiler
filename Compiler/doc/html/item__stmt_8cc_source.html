101 std::string compiler::Item_stmt_eif::print_result(const uint32_t& indent,

104  print_indent(indent, leaf, oss);
105  oss << " If-else Statement" << std::endl;
106  oss << condition->print_result(indent + 2, false);
107 
108  // There could be no else statement at all.
109  if (else_branch != nullptr) {
