101std::string compiler::Item_stmt_eif::print_result(const uint32_t& indent,
102 const bool& leaf) const {
103 std::ostringstream oss;
104 print_indent(indent, leaf, oss);
105 oss << " If-else Statement" << std::endl;
106 oss << condition->print_result(indent + 2, false);
107
108 // There could be no else statement at all.
109 if (else_branch != nullptr) {
110 oss << if_branch->print_result(indent + 2, false);
111 oss << else_branch->print_result(indent + 2, true);
112 } else {
113 oss << if_branch->print_result(indent + 2, true);
114 }
115 return oss.str();
116}

166std::string compiler::Item_block::print_result(const uint32_t& indent,
167 const bool& leaf) const {
168 std::ostringstream oss;
169 print_indent(indent, leaf, oss);
170 oss << " Block" << std::endl;
171 for (uint32_t i = 0; i < statements.size(); i++) {
172 oss << statements[i]->print_result(indent + 2, i == statements.size() - 1);
173 }
174 return oss.str();
175}
