72std::string
73compiler::Item_literal_numeric::print_result(const uint32_t& indent, const bool& leaf) const
74{
75 std::ostringstream oss;
76
77 print_indent(indent, leaf, oss);
78 oss << " Literal Numeric with value ";
79
80 switch (get_literal_type()) {
81 case Item_literal::literal_type::INT_TYPE: {
82 oss << int(value) << termcolor::reset << std::endl;
83 break;
84 }
85 case Item_literal::literal_type::CHAR_TYPE: {
86 oss << char(value) << termcolor::reset << std::endl;
87 break;
88 }
89 case Item_literal::literal_type::REAL_TYPE: {
90 oss << double(value) << termcolor::reset << std::endl;
91 break;
92 }
93 default: {
94 throw compiler::type_error("Wrong numeric type!");
95 }
96 }
97
98 return oss.str();
99}

111std::string
112compiler::Item_literal_array_init::print_result(const uint32_t& indent, const bool& leaf) const
113{
114 std::stringstream oss;
115 print_indent(indent, leaf, oss);
116 oss << " Literal Array Init" << std::endl;
117
118 if (expression != nullptr) {
119 oss << expression->print_result(indent + 2, false) << std::endl;
120 }
121
122 for (uint32_t i = 0; i < value_list.size(); i++) {
123 oss << value_list[i]->print_result(indent + 2, i == value_list.size() - 1);
124 }
125
126 return oss.str();
