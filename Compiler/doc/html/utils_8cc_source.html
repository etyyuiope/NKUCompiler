127void compiler::print_indent(const uint32_t& indent, const bool& leaf,
128 std::ostream& os) {
129 // Print indentation.
130 for (uint32_t i = 0; i < indent; i += 2) {
131 os << termcolor::bright_green << "│ " << termcolor::reset;
132 }
133
134 // Print node.
135 if (leaf) {
136 os << termcolor::bright_green << (std::string) "└──" << termcolor::reset;
137 } else {
138 os << termcolor::bright_green << (std::string) "├──" << termcolor::reset;
139 }
140}

142bool compiler::is_dir(const std::string& file_name) {
143 struct stat status;
144
145 if (stat(file_name.data(), &status) == 0) {
146 if (S_ISDIR(status.st_mode)) {
147 return true;
148 }
149 } else {
150 throw std::invalid_argument("Path does not exist!");
151 }
152
153 return false;
154}

156compiler::ir::var_type compiler::to_ir_type(const basic_type& b_type) {
157 switch (b_type) {
158 case basic_type::CHAR_TYPE:
159 return ir::var_type::i8;
160 case basic_type::INT_TYPE:
161 return ir::var_type::i32;
162 case basic_type::REAL_TYPE:
163 return ir::var_type::DB;
164 default:
165 throw compiler::unsupported_operation("Cannot convert this type to IR!");
166 }
