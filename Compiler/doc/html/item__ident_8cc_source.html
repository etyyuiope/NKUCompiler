73std::string compiler::Item_ident_array::print_result(const uint32_t& indent,
74 const bool& leaf) const {
75 std::ostringstream oss;
76
77 print_indent(indent, leaf, oss);
78 oss << " Array Identifier with name " << termcolor::red << name
79 << termcolor::reset << ", and the shape is " << std::endl;
80 for (uint32_t i = 0; i < array_shape.size(); i++) {
81 oss << array_shape[i]->print_result(
82 indent + 2,
83 i == array_shape.size() - 1); // Each number is a dimension.
84 }
85 return oss.str();
