21 void compiler::Item_func_def::generate_ir_helper(
22  compiler::ir::IRContext* const ir_context,
23  std::vector<compiler::ir::IR>& ir_list) const {
24  try {
25  ir_context->get_symbol_table()->add_symbol(
26  identifier->get_name(),
27  new compiler::Symbol(identifier->get_name(),
28  compiler::symbol_type::FUNC_TYPE));
29  ir_context->enter_scope();
30  const size_t argument_number = parameter->get_arg_number();
31 
32  // BEGIN_FUNCTION.
33  ir_list.emplace_back(
34  ir::op_type::BEGIN_FUNC, nullptr,
35  new ir::Operand(to_ir_type(return_type), "",
36  std::to_string(argument_number), false, false),
37  ir::global_sign + identifier->get_name());
38  ir_list.emplace_back(ir::op_type::LBL, ".LB_" + identifier->get_name() + "_BEGIN:");
39 
40  // Get all the arguments.
41  const std::vector<Item_func_def_arg*> arguments =
42  parameter->get_arguments();
43  for (size_t i = 0; i < argument_number; i++) {
44  // Allocate variable to store the arguments.
45  const compiler::Item_func_def_arg* const arg = arguments[i];
46  const compiler::Item_ident* const identifier = arg->get_identifier();
47 
48  // Handle array type.
49  // Currently we do not support non-array and non-variable type.
50  if (identifier->get_ident_type() == Item_ident::ARRAY) {
51  } else {
52  const std::string var =
53  ir::local_sign +
54  std::to_string(ir_context->get_symbol_table()->get_available_id());
55  ir_list.emplace_back(ir::op_type::MOV, new ir::Operand(var),
56  ir::arg_sign + std::to_string(i));
57  ir_context->get_symbol_table()->add_symbol(
58  identifier->get_name(),
59  new compiler::Symbol(var, compiler::symbol_type::VAR_TYPE));
60  }
61  }
62 
63  // Generate the IR for the function body.
64  func_body->generate_ir(ir_context, ir_list);
65  // No return type? Add it manually.
66  if (ir_list.back().get_op_type() != ir::op_type::RET) {
67  ir::Operand* return_value = nullptr;
68 
69  if (return_type == compiler::basic_type::VOID_TYPE) {
70  return_value =
71  new ir::Operand(ir::var_type::NONE, "", "", false, false);
72  } else {
73  return_value = new ir::Operand(compiler::to_ir_type(return_type), "",
74  "0", false, false);
75  }
76 
77  ir_list.emplace_back(ir::op_type::RET, return_value);
78  }
79 
80  ir_list.emplace_back(ir::op_type::END_FUNC);
81  ir_context->leave_scope();
82  } catch (const std::exception& e) {
83  std::cerr << termcolor::red << termcolor::bold << lineno << ": " << e.what()
84  << termcolor::reset << std::endl;
85  exit(1);
86  }
