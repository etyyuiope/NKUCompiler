20void compiler::Item_func_def::generate_ir_helper(
21 compiler::ir::IRContext* const ir_context,
22 std::vector<compiler::ir::IR>& ir_list) const {
23 ir_context->enter_scope();
24 const size_t argument_number = parameter->get_arg_number();
25
26 // BEGIN_FUNCTION.
27 ir_list.emplace_back(
28 ir::op_type::BEGIN_FUNC, nullptr,
29 new ir::Operand(to_ir_type(return_type), "", std::to_string(argument_number), false, false),
30 identifier->get_name());
31
32 // Get all the arguments.
33 const std::vector<Item_func_def_arg*> arguments = parameter->get_arguments();
34 for (size_t i = 0; i < argument_number; i++) {
35 const compiler::Item_func_def_arg* const arg = arguments[i];
36 const compiler::Item_ident* const identifier = arg->get_identifier();
37 }
38
39 ir_list.emplace_back(ir::op_type::END_FUNC, identifier->get_name());
40 ir_context->leave_scope();
41}

43compiler::ir::Operand* compiler::Item_func_call::eval_runtime_helper(
44 compiler::ir::IRContext* const ir_context,
45 std::vector<compiler::ir::IR>& ir_list) const {
46 std::vector<ir::Operand*> operands;
47 ir::Operand* const dst = new ir::Operand(
48 "%" +
49 std::to_string(ir_context->get_symbol_table()->get_top_scope_uuid()));
50
51 const std::vector<compiler::Item_expr*> args = arguments->get_arguments();
52 for (uint32_t i = 0; i < args.size(); i++) {
53 operands.emplace_back(args[i]->eval_runtime(ir_context, ir_list));
54 }
55
56 // Set arguments.
57 for (uint32_t i = 0; i < args.size(); i++) {
58 ir::Operand* const arg_pos = new ir::Operand(ir::var_type::NONE, std::to_string(i), "", false, false);
59 ir_list.emplace_back(ir::op_type::ARG, arg_pos, identifier->get_name());
60 }
61
62 ir_list.emplace_back(ir::op_type::CALL, dst, identifier->get_name());
63 return dst;
64}
